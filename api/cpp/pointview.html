void PointView::setFieldInternal(Dimension::Id dim, PointId idx, const void *value) { PointId rawId = 0; if (idx == size()) { rawId = m_pointTable.addPoint(); m_index.push_back(rawId); m_size++; assert(m_temps.empty()); } else if (idx > size()) { std::cerr << “Point index must increment.\n”; error - throw? return; } else { rawId = m_index[idx]; } m_pointTable.setFieldInternal(dim, rawId, value); }
